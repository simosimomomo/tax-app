<script>

    let taxChartInstance = null;


    function calculateTaxes() {

        const income = parseInt(document.getElementById('income').value) || 0;

        const spending = parseInt(document.getElementById('spending').value) || 0;


        if (income === 0) {

            alert('年収を入力してください。');

            return;

        }


        // 1. 社会保険料の概算 (約15%)

        const socialInsurance = Math.floor(income * 0.15);


        // 2. 給与所得控除の概算計算

        let incomeDeduction = 0;

        if (income <= 1625000) incomeDeduction = 550000;

        else if (income <= 1800000) incomeDeduction = income * 0.4 - 100000;

        else if (income <= 3600000) incomeDeduction = income * 0.3 + 80000;

        else if (income <= 6600000) incomeDeduction = income * 0.2 + 440000;

        else if (income <= 8500000) incomeDeduction = income * 0.1 + 1100000;

        else incomeDeduction = 1950000;


        // 3. 課税所得の概算 (基礎控除を所得税48万、住民税43万とする)

        let taxableIncomeTax = income - socialInsurance - incomeDeduction - 480000;

        if (taxableIncomeTax < 0) taxableIncomeTax = 0;

        

        let taxableResidentTax = income - socialInsurance - incomeDeduction - 430000;

        if (taxableResidentTax < 0) taxableResidentTax = 0;


        // 4. 所得税の概算

        let incomeTax = 0;

        if (taxableIncomeTax <= 1949000) incomeTax = taxableIncomeTax * 0.05;

        else if (taxableIncomeTax <= 3299000) incomeTax = taxableIncomeTax * 0.10 - 97500;

        else if (taxableIncomeTax <= 6949000) incomeTax = taxableIncomeTax * 0.20 - 427500;

        else if (taxableIncomeTax <= 8999000) incomeTax = taxableIncomeTax * 0.23 - 636000;

        else if (taxableIncomeTax <= 17999000) incomeTax = taxableIncomeTax * 0.33 - 1536000;

        else incomeTax = taxableIncomeTax * 0.40 - 2796000;

        incomeTax = Math.floor(incomeTax);


        // 5. 住民税の概算 (約10% + 均等割5000円)

        const residentTax = taxableResidentTax > 0 ? Math.floor(taxableResidentTax * 0.10) + 5000 : 0;


        // 6. 消費税の概算 (支出は税込と仮定し、10%計算：支出 × 10/110)

        // ※軽減税率(8%)は考慮せず一律10%で計算

        const consumptionTax = Math.floor(spending * (10 / 110));

        const pureSpending = spending - consumptionTax; // 税抜の支出


        // 7. 手元に残るお金 (貯金等)

        let remainingMoney = income - socialInsurance - incomeTax - residentTax - spending;

        if (remainingMoney < 0) remainingMoney = 0;


        // 総負担額

        const totalTaxes = socialInsurance + incomeTax + residentTax + consumptionTax;


        // UIの更新

        updateUI(income, socialInsurance, incomeTax, residentTax, consumptionTax, pureSpending, remainingMoney, totalTaxes);

        drawChart(socialInsurance, incomeTax, residentTax, consumptionTax, pureSpending, remainingMoney);

    }


    function updateUI(income, soc, incTax, resTax, consTax, spend, remain, total) {

        document.getElementById('resultSection').style.display = 'flex';

        

        const detailsHtml = `

            <h3>内訳（年間概算）</h3>

            <div class="detail-item"><span>額面年収:</span> <span>${income.toLocaleString()} 円</span></div>

            <div class="detail-item"><span>社会保険料:</span> <span>${soc.toLocaleString()} 円</span></div>

            <div class="detail-item"><span>所得税:</span> <span>${incTax.toLocaleString()} 円</span></div>

            <div class="detail-item"><span>住民税:</span> <span>${resTax.toLocaleString()} 円</span></div>

            <div class="detail-item"><span>消費税:</span> <span>${consTax.toLocaleString()} 円</span></div>

            <div class="detail-item"><span>その他の支出(税抜):</span> <span>${spend.toLocaleString()} 円</span></div>

            <div class="detail-item"><span>手元に残るお金(貯蓄等):</span> <span>${remain.toLocaleString()} 円</span></div>

            <div class="detail-item total"><span>国・自治体に納めた総額:</span> <span>${total.toLocaleString()} 円</span></div>

        `;

        document.getElementById('detailsList').innerHTML = detailsHtml;

    }


    function drawChart(soc, incTax, resTax, consTax, spend, remain) {

        const ctx = document.getElementById('taxChart').getContext('2d');

        

        // 既存のグラフがあれば破棄して再描画

        if (taxChartInstance) {

            taxChartInstance.destroy();

        }


        taxChartInstance = new Chart(ctx, {

            type: 'pie',

            data: {

                labels: ['社会保険料', '所得税', '住民税', '消費税', 'その他の支出(税抜)', '手元に残るお金'],

                datasets: [{

                    data: [soc, incTax, resTax, consTax, spend, remain],

                    backgroundColor: [

                        '#e74c3c', // 赤: 社会保険

                        '#e67e22', // オレンジ: 所得税

                        '#f1c40f', // 黄色: 住民税

                        '#9b59b6', // 紫: 消費税

                        '#95a5a6', // 灰: 支出

                        '#2ecc71'  // 緑: 残り

                    ],

                    borderWidth: 1

                }]

            },

            options: {

                responsive: true,

                maintainAspectRatio: false,

                plugins: {

                    legend: {

                        position: 'bottom',

                    },

                    tooltip: {

                        callbacks: {

                            label: function(context) {

                                let label = context.label || '';

                                if (label) {

                                    label += ': ';

                                }

                                if (context.parsed !== null) {

                                    label += new Intl.NumberFormat('ja-JP').format(context.parsed) + ' 円';

                                }

                                return label;

                            }

                        }

                    }

                }

            }

        });

    }
